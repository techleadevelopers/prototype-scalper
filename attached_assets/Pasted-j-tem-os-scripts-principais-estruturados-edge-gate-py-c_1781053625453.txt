Se o mercado entrar em um dia de calmaria extrema, 10% de distância fará o robô passar dias sem executar nada. Se entrar em dia de colapso macro, 10% pode ser rompido como papel.O algoritmo vai misturar os seus alvos com o ATR de 15m:LONG (Modo Queda Livre): Inicia a escada a partir de $10\%$ abaixo do preço atual (ou baseado em $3.5 \times \text{ATR}_{15m}$ para se ajustar à volatilidade real). Cada degrau subsequente desce $1\%$.SHORT (Modo Parabólico): Inicia a rede a partir de $20\%$ acima do preço atual ($5.0 \times \text{ATR}_{15m}$), pulverizando ordens acima com espaçamento de $1\%$ para capturar a exaustão de compra.📑 II. def build_sniper_tail_grid(symbol: str, current_price: float, side: str, base_target_usdt: float):
    """
    Gera uma malha/escada de gatilhos na pedra em massa para capturar pavios extremos (espetos).
    LONG começa em -10% | SHORT começa em +20% do preço atual.
    """
    grid_layers = []
    
    if side == "LONG":
        # Definição dos multiplicadores de queda especificados por você
        drop_percentages = [0.10, 0.11, 0.12] 
        # Distribuição de lote por degrau (Martingale controlado ou linear)
        allocation_weights = [0.20, 0.30, 0.50] 
        
        for idx, drop in enumerate(drop_percentages):
            trigger_p = current_price * (1.0 - drop)
            # Alvo curto dinâmico escalado com base na Feature 5 do Target Scaler
            tp_pct = (base_target_usdt / trigger_p) * 100
            sl_pct = tp_pct * 2 # Relação controlada sobre a margem isolada
            
            grid_layers.append({
                "level": idx + 1,
                "side": "LONG",
                "triggerPrice": round(trigger_p, 4),
                "targetPrice": round(trigger_p * (1.0 + (tp_pct / 100)), 4),
                "stopPrice": round(trigger_p * (1.0 - (sl_pct / 100)), 4),
                "allocationFactor": allocation_weights[idx]
            })
            
    elif side == "SHORT":
        # Definição dos multiplicadores de subida especificados por você
        pump_percentages = [0.20, 0.21, 0.22, 0.24]
        allocation_weights = [0.15, 0.25, 0.30, 0.30]
        
        for idx, pump in enumerate(pump_percentages):
            trigger_p = current_price * (1.0 + pump)
            tp_pct = (base_target_usdt / trigger_p) * 100
            sl_pct = tp_pct * 2
            
            grid_layers.append({
                "level": idx + 1,
                "side": "SHORT",
                "triggerPrice": round(trigger_p, 4),
                "targetPrice": round(trigger_p * (1.0 - (tp_pct / 100)), 4),
                "stopPrice": round(trigger_p * (1.0 + (sl_pct / 100)), 4),
                "allocationFactor": allocation_weights[idx]
            })

    return grid_layers

import { bingxClient } from "./bingxClient";

export async function deploySniperGridInMassa(symbol: string, brainGridResponse: any): Promise<void> {
  const { grid, executionMetrics } = brainGridResponse;
  
  console.log(`[🎯 BATCH SNIPER] Armanda rede de captura para ${symbol}. Total de degraus: ${grid.length}`);

  // Dispara todas as ordens em paralelo no barramento HTTP para diminuir o tempo de execução
  const gridPromises = grid.map(async (layer: any) => {
    try {
      const order = await bingxClient.placeOrder({
        symbol: symbol,
        side: layer.side === "LONG" ? "BUY" : "SELL",
        type: "LIMIT",
        price: layer.triggerPrice,
        quantity: (process.env.BASE_LOT_SIZE_USDT || 10) * layer.allocationFactor,
        postOnly: true, // Estrito: Garante taxa Maker e proíbe agredir o book antes do espeto
        takeProfit: layer.targetPrice,
        stopLoss: layer.stopPrice
      });
      
      console.log(`[✅ LAYER UNLOCKED] Degrau ${layer.level} posicionado na pedra: $${layer.triggerPrice}`);
      return order.id;
    } catch (err) {
      console.error(`[❌ LAYER FAILED] Falha ao injetar degrau ${layer.level} de ${symbol}:`, err);
      return null;
    }
  });

  const deployedOrderIds = await Promise.all(gridPromises);

  // Configura a expiração coletiva (TTL) do grid para limpar ordens fantasmas da pedra
  setTimeout(async () => {
    console.log(`[⏱️ GRID TTL] Verificando e limpando ordens remanescentes do grid de ${symbol}...`);
    await bingxClient.cancelAllOpenOrdersForSymbol(symbol, deployedOrderIds.filter(Boolean));
  }, executionMetrics.expirationSeconds * 1000);
}
