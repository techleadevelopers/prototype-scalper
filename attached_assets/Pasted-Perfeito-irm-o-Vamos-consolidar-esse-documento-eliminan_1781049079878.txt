🎛️ IV. Validação de Infraestrutura e Execução Rígida (Backend Node.js)O backend recebe o payload do Quant Brain e processa os filtros de segurança física da conta antes de submeter a ordem limitada para a BingX Exchange Futuros.TypeScript// backend/src/lib/triggerEngine.ts
import { quantBrainClient } from "./quantBrainClient";
import { bingxClient } from "./bingxClient";
import { positionSizing } from "./positionSizing";

export async function processMultiFrameTrigger(symbol: string, candleData: any): Promise<void> {
  // Solicita o veredito geométrico completo direto do Quant Brain
  const brainResponse = await quantBrainClient.evaluateEdge(symbol, candleData);
  
  if (brainResponse.decision !== "ARM_TRIGGER") return;

  // 1. CHECAGEM DE EXPOSIÇÃO EM MASSA: Evitar mais de 3 trades concomitantes
  const activePositionsCount = await bingxClient.getActivePositionsCount();
  if (activePositionsCount >= 3) {
    console.log(`[🚫 RISK BLOCK] Ordem descartada. Teto de 3 posições abertas atingido.`);
    return;
  }

  // 2. CHECAGEM DE LIQUIDEZ E SPREAD: Bloqueia se o book estiver aberto/inconsistente
  const orderBook = await bingxClient.getOrderBook(symbol);
  const spreadPct = (orderBook.asks[0].price - orderBook.bids[0].price) / orderBook.bids[0].price;
  if (spreadPct > 0.0002) { // Máximo de 0.02% de Spread admitido para alvos curtos
    console.log(`[🚫 LIQUIDITY BLOCK] Spread muito largo (${(spreadPct*100).toFixed(4)}%) em ${symbol}.`);
    return;
  }

  // 3. CALCULO DO LOTE: Usa a fração de Kelly injetada pelo cérebro quantitativo
  const orderSize = positionSizing.calculateByKelly({
    kellyFraction: brainResponse.kellyFraction,
    expectedValue: brainResponse.expectedValue
  });

  // 4. ENVIO DA ORDEM MAKER LIMIT: Post-Only obrigatório para eliminar taxas pesadas e slippage
  const orderResult = await bingxClient.placeOrder({
    symbol: symbol,
    side: brainResponse.side === "LONG" ? "BUY" : "SELL",
    type: "LIMIT",
    price: brainResponse.triggerPrice,
    quantity: orderSize,
    postOnly: true, // Garante execução estritamente como Maker na BingX
    takeProfit: brainResponse.targetPrice,
    stopLoss: brainResponse.stopPrice
  });

  // 5. CANCELAMENTO RELÂMPAGO: Evita ordens fantasmas pendentes na pedra se o mercado mudar de contexto
  setTimeout(async () => {
    const status = await bingxClient.getOpenOrderStatus(symbol, orderResult.id);
    if (status === "UNFILLED") {
      await bingxClient.cancelOpenOrder(symbol, orderResult.id);
      console.log(`[⏱️ STALE EXPIRY] Ordem limitada expirada e removida da pedra em ${symbol}.`);
    }
  }, brainResponse.expirationSeconds * 1000);
}
