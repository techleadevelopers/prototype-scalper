Para moedas pesadas (como BTC ou ETH), 10% é um movimento de crash de mercado.A Regra de Ouro para o edge_gate.py:Para manter o sistema estável e contínuo, a IA deve ler o parâmetro base do seu .env (ex: GRID_START_LONG_PCT=10), mas cruzar com o ATR de 15m para garantir que o preço não mude de estrutura antes de buscar a ordem.Para LONG (Alvo na Queda): O sistema pega o preço atual e calcula níveis subsequentes (ex: $10\%, 11\%, 12\%$) injetando as frações de volume.Para SHORT (Alvo na Exaustão de Subida): O sistema joga a rede lá em cima ($20\%, 21\%, 22\%$), esperando o esgotamento do volume comprador.📑 II. 🎛️ III. Implementação no Backend Executor (Node.js)O backend recebe esse payload do tipo ARM_TRIGGER_GRID e, em vez de enviar uma única ordem, faz um disparo em lote (Batch Order Placement) na BingX, usando a flag postOnly: true para garantir que todas entrem como Maker.TypeScript// backend/src/lib/gridTriggerManager.ts
import { bingxClient } from "./bingxClient";
import { positionSizing } from "./positionSizing";

export async function processGridTrigger(symbol: string, brainResponse: any) {
  if (brainResponse.decision !== "ARM_TRIGGER_GRID") return;

  // 1. Calcula o tamanho total da mão para a operação de Grid
  const totalSize = positionSizing.getGridBaseSize(symbol);

  // 2. Loop para espalhar as ordens na pedra em massa
  for (const layer of brainResponse.grid) {
    try {
      // Modula o tamanho do lote de cada degrau com base no allocationFactor da IA
      const layerSize = totalSize * layer.allocationFactor;

      await bingxClient.placeOrder({
        symbol: symbol,
        side: layer.side === "LONG" ? "BUY" : "SELL",
        type: "LIMIT",
        price: layer.triggerPrice,
        quantity: layerSize,
        postOnly: true, // Obrigatório: Garante taxa maker e não agride o book
        takeProfit: layer.targetPrice,
        stopLoss: layer.stopPrice
      });

      console.log(`[🎯 GRID SNIPER] Degrau ${layer.level} armado na pedra em ${symbol} a $${layer.triggerPrice}`);
    } catch (error) {
      console.error(`[❌ GRID ERROR] Falha ao armar camada ${layer.level} de ${symbol}:`, error);
    }
  }

  // 3. Monitoramento de Expiração Coletiva (TTL do Grid)
  setGridTimeout(symbol, brainResponse.executionMetrics.expirationSeconds);
}
