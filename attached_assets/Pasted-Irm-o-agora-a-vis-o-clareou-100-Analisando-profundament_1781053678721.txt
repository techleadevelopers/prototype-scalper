from __future__ import annotations

import time

# Função cirúrgica para gerar os múltiplos gatilhos (Grid) solicitados
def build_tail_hunter_grid(symbol: str, current_price: float, side: str, base_target_usdt: float) -> list[dict[str, Any]]:
    """
    Gera uma malha/escada de ordens limite (Maker) baseada na distância do preço atual.
    LONG: 10%, 11%, 12% abaixo do preço.
    SHORT: 20%, 21%, 22%, 24% acima do preço.
    """
    grid_layers = []
    
    if side == "LONG":
        # Percentuais de queda e pesos de alocação de margem
        drop_pcts = [0.10, 0.11, 0.12]
        weights = [0.20, 0.30, 0.50] # Distribuição do tamanho do lote (Linear-Martingale)
        
        for idx, drop in enumerate(drop_pcts):
            trigger_p = current_price * (1.0 - drop)
            # Dynamic Target Scaler (Feature 5): Transforma $0.50 em percentual dinâmico
            tp_pct = (base_target_usdt / trigger_p) * 100
            sl_pct = tp_pct * 2 # Relação matemática controlada de Risco/Retorno 2:1
            
            grid_layers.append({
                "level": idx + 1,
                "side": "LONG",
                "triggerPrice": round(trigger_p, 5),
                "targetPrice": round(trigger_p * (1.0 + (tp_pct / 100)), 5),
                "stopPrice": round(trigger_p * (1.0 - (sl_pct / 100)), 5),
                "allocationFactor": weights[idx]
            })
            
    elif side == "SHORT":
        # Percentuais de subida e pesos de alocação de margem
        pump_pcts = [0.20, 0.21, 0.22, 0.24]
        weights = [0.15, 0.25, 0.30, 0.30]
        
        for idx, pump in enumerate(pump_pcts):
            trigger_p = current_price * (1.0 + pump)
            tp_pct = (base_target_usdt / trigger_p) * 100
            sl_pct = tp_pct * 2
            
            grid_layers.append({
                "level": idx + 1,
                "side": "SHORT",
                "triggerPrice": round(trigger_p, 5),
                "targetPrice": round(trigger_p * (1.0 - (tp_pct / 100)), 5),
                "stopPrice": round(trigger_p * (1.0 + (sl_pct / 100)), 5),
                "allocationFactor": weights[idx]
            })
            
    return grid_layers

🧠 ELEVAÇÃO 2: offline_learner.py — Integrado ao Banco de Dados do Core (knowledge.db)
Chega de arquivos soltos. O modelo de inteligência quantitativa do robô deve ler diretamente a tabela trade_outcomes e a tabela signal_tracking que o seu knowledge_base.py já gerencia nativamente.

O script abaixo extrai as linhas purificadas e executa o retreino do modelo ajustando os pesos binários:

Python
# ==============================================================================
# NOVO MÓDULO: core/offline_learner.py
# ==============================================================================
from __future__ import annotations
import asyncio
import os
import time
import logging
from core import knowledge_base as kb
from core.shadow_model import train_shadow_model, shadow_model_status

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger("OfflineLearner")

class SniperOfflineLearner:
    def __init__(self):
        self.min_samples_required = 50

    async def run_24h_recalibration(self) -> dict[str, Any]:
        logger.info("[🧠 LEARNER] Iniciando ciclo autônomo de recalibração de 24h...")
        
        # Conecta no seu banco de dados nativo através do core
        async with kb.connect(kb.DB_PATH) as db:
            # Seleciona apenas os sinais purificados que viraram ordens executadas reais
            cursor = await db.execute("""
                SELECT s.source_id, s.symbol, s.features_json, t.realized_pnl 
                FROM signal_tracking s
                JOIN trade_outcomes t ON s.source_id = t.source_id
                WHERE t.realized_pnl IS NOT NULL
            """)
            rows = await cursor.fetchall()
            
        if len(rows) < self.min_samples_required:
            logger.warning(f"[⚠️ LEARNER] Amostragem insuficiente no banco de dados ({len(rows)}/{self.min_samples_required}). Re-treino abortado.")
            return {"status": "SKIPPED", "reason": "Insuficient samples"}

        logger.info(f"[🔬 LEARNER] Processando {len(rows)} amostras de alta fidelidade para ajuste de pesos...")
        
        try:
            # Dispara a função nativa do seu core/shadow_model.py para retreinar
            # Isso garante total estabilidade e zero conflito de concorrência
            start_time = time.time()
            
            # Executa o retreino nativo do sistema passando a chave do pipeline
            await train_shadow_model() 
            
            duration = round(time.time() - start_time, 2)
            model_info = shadow_model_status()
            
            logger.info(f"[✅ LEARNER] Recalibrado com sucesso em {duration}s! Novo modelo gerado.")
            return {
                "status": "SUCCESS",
                "duration_seconds": duration,
                "samples_trained": len(rows),
                "model_status": model_info
            }
        except Exception as e:
            logger.error(f"[❌ LEARNER] Erro crítico durante o treinamento do modelo: {str(e)}")
            return {"status": "FAILED", "error": str(e)}

if __name__ == "__main__":
    learner = SniperOfflineLearner()
    asyncio.run(learner.run_24h_recalibration())
