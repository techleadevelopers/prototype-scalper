import { quantBrainClient } from "./quantBrainClient";
import { bingxClient } from "./bingxClient";

interface MuxLockState {
  isActive: boolean;
  expiresAt: number;
  triggeringSignalId: string | null;
}

class BatchTriggerOrchestrator {
  private muxLock: MuxLockState = { isActive: false, expiresAt: 0, triggeringSignalId: null };
  private activePositionsCount: number = 0;
  private maxConcurrentPositions: number = 3;

  constructor() {
    const candlesLock = Number(process.env.MUX_LOCK_CANDLES) || 2;
    console.log(`[🎯 BATCH] Orquestrador iniciado. MUX_LOCK_CANDLES = ${candlesLock}m`);
  }

  /**
   * Avalia e processa o sinal para um símbolo da lista do .env
   */
  public async handleMarketTick(symbol: string, candleData: any): Promise<void> {
    const now = Date.now();

    // 1. HARD GATE: Verifica se o Mux-Lock global está ativo
    if (this.muxLock.isActive) {
      if (now < this.muxLock.expiresAt) {
        const remainingSec = Math.ceil((this.muxLock.expiresAt - now) / 1000);
        // Log silencioso ou debug para não poluir o painel
        return; 
      } else {
        // Mux-Lock expirou naturalmente, libera o barramento
        this.muxLock.isActive = false;
        this.muxLock.triggeringSignalId = null;
        console.log(`[🔓 MUX_LOCK] Barramento liberado automaticamente.`);
      }
    }

    // 2. HARD GATE: Valida o teto de posições abertas em massa simultâneas
    this.activePositionsCount = await bingxClient.getActivePositionsCount();
    if (this.activePositionsCount >= this.maxConcurrentPositions) {
      return; // Aborta sem processar para economizar chamadas de API
    }

    // 3. EXECUTA O PIPELINE DO EDGE (Quant Brain Python)
    const brainResponse = await quantBrainClient.evaluateEdge(symbol, candleData);
    if (brainResponse.decision !== "ARM_TRIGGER") return;

    // 4. ATIVAÇÃO DO GATILHO E DISPARO DO MUX-LOCK SÍNCRONO
    await this.armTriggerAndLock(symbol, brainResponse);
  }

  private async armTriggerAndLock(symbol: string, signal: any): Promise<void> {
    try {
      // Envia a ordem limitada Maker (Post-Only) para a pedra na BingX
      const order = await bingxClient.placeOrder({
        symbol: symbol,
        side: signal.geometry.side === "LONG" ? "BUY" : "SELL",
        type: "LIMIT",
        price: signal.geometry.triggerPrice,
        takeProfit: signal.geometry.targetPrice,
        stopLoss: signal.geometry.stopPrice,
        postOnly: true
      });

      // Ativa o bloqueio de disparo por tick imediatamente (Feature 1)
      const lockDurationMs = (Number(process.env.MUX_LOCK_CANDLES) || 2) * 60 * 1000;
      this.muxLock = {
        isActive: true,
        expiresAt: Date.now() + lockDurationMs,
        triggeringSignalId: signal.metadata.signalId
      };

      console.log(`[🔒 MUX_LOCK_ACTIVE] Gatilho armado em ${symbol}. Bloqueio global ativado por ${lockDurationMs / 1000}s.`);
      
      // Inicializa o background poll de 5s para monitorar o preenchimento ou expiração (TTL)
      this.startTriggerLifecycleMonitor(symbol, order.id, signal);

    } catch (error) {
      console.error(`[❌ BATCH_ERROR] Falha ao armar gatilho para ${symbol}:`, error);
    }
